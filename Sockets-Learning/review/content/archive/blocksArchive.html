Explicación más profunda

Construimos el sistema en 5 capas de complejidad, desde la lectura/escritura básica hasta la seguridad completa con cifrado, checksum y compresión.

1

Lectura / escritura básica de bloques

Lo más simple: leer y escribir archivos en bloques

La base de todo el sistema. Leemos un archivo binario en trozos de tamaño fijo (512 bytes por defecto) y los escribimos secuencialmente en un destino. Sin metadatos, sin validación: solo transferencia pura.

Python 
# Lo más básico: leer y escribir archivos en bloques
def capa1_leer_escribir(origen, destino, tam_bloque=512):
    with open(origen, 'rb') as f_origen:
        with open(destino, 'wb') as f_destino:
            while True:
                bloque = f_origen.read(tam_bloque)
                if not bloque:
                    break
                f_destino.write(bloque)
2

Dividir en bloques y guardar metadata

Cada bloque como archivo separado + índice JSON

Aquí introducimos la metadata: cada bloque se guarda como un archivo independiente (.bin) y se genera un índice en formato JSON con número, tamaño y ruta de cada bloque. Esto permite gestionarlos individualmente.

Python 
# Guardar cada bloque como archivo separado + índice
def capa2_dividir_en_bloques(archivo, tam_bloque=512, carpeta_destino="bloques"):
    import os
    os.makedirs(carpeta_destino, exist_ok=True)

    with open(archivo, 'rb') as f:
        num_bloque = 0
        metadata = []

        while True:
            bloque = f.read(tam_bloque)
            if not bloque:
                break

            # Guardar cada bloque como archivo independiente
            nombre_bloque = f"{carpeta_destino}/bloque_{num_bloque:04d}.bin"
            with open(nombre_bloque, 'wb') as f_bloque:
                f_bloque.write(bloque)

            # Guardar metadata del bloque
            metadata.append({
                'numero': num_bloque,
                'tamano': len(bloque),
                'archivo': nombre_bloque
            })

            num_bloque += 1

    # Guardar índice (metadata) como JSON
    import json
    with open(f"{carpeta_destino}/metadata.json", 'w') as f:
        json.dump(metadata, f, indent=2)

    print(f"✅ Dividido en {num_bloque} bloques")
    return metadata
3

Reconstruir desde bloques

Leer metadata, ordenar y recomponer el archivo original

El proceso inverso: cargamos el JSON de metadata, ordenamos los bloques por número y los concatenamos en orden para reconstruir el archivo binario de salida.

Python 
def capa3_reconstruir_desde_bloques(carpeta="bloques", archivo_salida="reconstruido.bin"):
    import json

    # Leer metadata
    with open(f"{carpeta}/metadata.json", 'r') as f:
        metadata = json.load(f)

    # Ordenar bloques por número
    metadata_ordenada = sorted(metadata, key=lambda x: x['numero'])

    # Reconstruir archivo
    with open(archivo_salida, 'wb') as f_salida:
        for info in metadata_ordenada:
            with open(info['archivo'], 'rb') as f_bloque:
                bloque = f_bloque.read()
                f_salida.write(bloque)

    print(f"✅ Archivo reconstruido: {archivo_salida}")
4

Verificación de integridad (checksums)

Añadimos hash SHA-256 para detectar bloques corruptos

Cada bloque ahora incluye un checksum SHA-256 en su metadata. Antes de reconstruir, verificamos que el hash calculado coincida con el esperado. Si hay discrepancias, se reportan los bloques corruptos.

Python 
import hashlib

def capa4_guardar_con_checksum(archivo, tam_bloque=512, carpeta="bloques_seguros"):
    import os, json
    os.makedirs(carpeta, exist_ok=True)

    with open(archivo, 'rb') as f:
        num_bloque = 0
        metadata = []

        while True:
            bloque = f.read(tam_bloque)
            if not bloque:
                break

            # Calcular checksum (hash) del bloque
            checksum = hashlib.sha256(bloque).hexdigest()

            # Guardar bloque
            nombre_bloque = f"{carpeta}/bloque_{num_bloque:04d}.bin"
            with open(nombre_bloque, 'wb') as f_bloque:
                f_bloque.write(bloque)

            # Guardar metadata con checksum
            metadata.append({
                'numero': num_bloque,
                'tamano': len(bloque),
                'archivo': nombre_bloque,
                'checksum': checksum  # ← NUEVO: para verificar integridad
            })

            num_bloque += 1

    with open(f"{carpeta}/metadata.json", 'w') as f:
        json.dump(metadata, f, indent=2)

    return metadata


def capa4_verificar_bloques(carpeta="bloques_seguros"):
    import json
    errores = []

    with open(f"{carpeta}/metadata.json", 'r') as f:
        metadata = json.load(f)

    for info in metadata:
        with open(info['archivo'], 'rb') as f_bloque:
            bloque = f_bloque.read()
            checksum_calculado = hashlib.sha256(bloque).hexdigest()

            if checksum_calculado != info['checksum']:
                errores.append({
                    'bloque': info['numero'],
                    'esperado': info['checksum'],
                    'obtenido': checksum_calculado
                })

    if errores:
        print(f"❌ Se encontraron {len(errores)} bloques corruptos")
        for error in errores:
            print(f"  - Bloque {error['bloque']}: corrupto")
        return False
    else:
        print("✅ Todos los bloques están integros")
        return True
5

Seguridad completa: Cifrado + Checksum + Compresión

Fernet (AES-128) · zlib · SHA-256 · Máxima protección

La capa definitiva. Cada bloque pasa por tres etapas: compresión → cifrado → checksum. Usamos cryptography.fernet para cifrado simétrico, zlib para compresión y hashlib para integridad. La clave se almacena aparte y se verifica cada bloque antes de descifrar.

Python 
from cryptography.fernet import Fernet
import zlib
import base64

def capa5_seguridad_completa(archivo, tam_bloque=512, carpeta="bloques_cifrados"):
    import os, json

    # Generar clave de cifrado
    clave = Fernet.generate_key()
    cipher = Fernet(clave)

    # Guardar clave (en producción esto iría en un lugar seguro)
    with open(f"{carpeta}/clave.key", 'wb') as f:
        f.write(clave)

    os.makedirs(carpeta, exist_ok=True)
    metadata = []

    with open(archivo, 'rb') as f:
        num_bloque = 0

        while True:
            bloque = f.read(tam_bloque)
            if not bloque:
                break

            # PASO 1: Comprimir el bloque
            bloque_comprimido = zlib.compress(bloque, level=6)

            # PASO 2: Cifrar el bloque comprimido
            bloque_cifrado = cipher.encrypt(bloque_comprimido)

            # PASO 3: Calcular checksum del bloque cifrado
            checksum = hashlib.sha256(bloque_cifrado).hexdigest()

            # Guardar bloque cifrado
            nombre_bloque = f"{carpeta}/bloque_{num_bloque:04d}.enc"
            with open(nombre_bloque, 'wb') as f_bloque:
                f_bloque.write(bloque_cifrado)

            metadata.append({
                'numero': num_bloque,
                'tamano_original': len(bloque),
                'tamano_comp_cif': len(bloque_cifrado),
                'archivo': nombre_bloque,
                'checksum': checksum
            })

            num_bloque += 1

    with open(f"{carpeta}/metadata.json", 'w') as f:
        json.dump(metadata, f, indent=2)

    print(f"✅ Seguridad aplicada: {num_bloque} bloques cifrados")
    print(f"🔑 Clave guardada en: {carpeta}/clave.key")
    return metadata


def capa5_reconstruir_seguro(carpeta="bloques_cifrados", archivo_salida="reconstruido_seguro.bin"):
    import json
    from cryptography.fernet import Fernet
    import zlib

    # Cargar clave
    with open(f"{carpeta}/clave.key", 'rb') as f:
        clave = f.read()
    cipher = Fernet(clave)

    # Cargar metadata
    with open(f"{carpeta}/metadata.json", 'r') as f:
        metadata = json.load(f)

    # Verificar integridad de todos los bloques antes de descifrar
    for info in metadata:
        with open(info['archivo'], 'rb') as f_bloque:
            bloque_cifrado = f_bloque.read()
            checksum_calculado = hashlib.sha256(bloque_cifrado).hexdigest()

            if checksum_calculado != info['checksum']:
                raise Exception(f"⚠️ Bloque {info['numero']} corrupto! No se puede reconstruir.")

    # Reconstruir
    with open(archivo_salida, 'wb') as f_salida:
        for info in sorted(metadata, key=lambda x: x['numero']):
            with open(info['archivo'], 'rb') as f_bloque:
                bloque_cifrado = f_bloque.read()

                # Descifrar
                bloque_comprimido = cipher.decrypt(bloque_cifrado)

                # Descomprimir
                bloque_original = zlib.decompress(bloque_comprimido)

                f_salida.write(bloque_original)

    print(f"✅ Archivo seguro reconstruido: {archivo_salida}")
